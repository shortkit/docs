The ShortKit Web SDK gives your web app a full-screen, vertically swipeable short-form video feed. It works as a React component or mounts directly to any DOM element, with HLS streaming handled internally.

## Requirements

| Parameter | Value |
|---|---|
| Chrome | 80+ |
| Firefox | 75+ |
| Safari | 14+ |
| Edge | 80+ |
| iOS Safari | 14+ |
| Chrome Android | 80+ |
| React (optional) | 18+ |

<Note>
  Internet Explorer is not supported.
</Note>

## Initialization

Create a `ShortKit` instance with your publishable key and config. Do this once, before mounting any feed.

<Tabs>
  <Tab title="ES Module">
    ```typescript main.ts
    import { ShortKit } from '@shortkit/web';

    const shortKit = ShortKit.init({
      apiKey: 'pk_live_your_publishable_key',
      config: {
        feedHeight: 'fullscreen',
        controls: {
          playPause: true,
          scrubber: true,
          captions: { enabled: true, defaultOn: false },
          volume: true,
          muteOnStart: true,
          share: { enabled: true },
        },
        theme: {
          primaryColor: '#FFFFFF',
          controlsTint: '#FFFFFF',
          backgroundColor: '#000000',
        },
      },
      environment: 'production',
    });
    ```
  </Tab>
  <Tab title="CDN">
    ```html
    <script src="https://cdn.shortkit.dev/web/v1/shortkit.min.js"></script>
    <script>
      const shortKit = ShortKit.init({
        apiKey: 'pk_live_your_publishable_key',
        config: {
          feedHeight: 'fullscreen',
          controls: {
            playPause: true,
            scrubber: true,
            volume: true,
            muteOnStart: true,
          },
          theme: {
            primaryColor: '#FFFFFF',
            backgroundColor: '#000000',
          },
        },
      });
    </script>
    ```
  </Tab>
</Tabs>

<Warning>
  Never expose your secret key (`sk_live_...` or `sk_test_...`) in client-side code, source control, or your app bundle. Use only your publishable key (`pk_live_...`) for SDK initialization.
</Warning>

## Displaying the feed

<Tabs>
  <Tab title="React">
    Use the `ShortKitFeed` React component:

    ```tsx VideoFeed.tsx
    import React from 'react';
    import { ShortKitFeed } from '@shortkit/web/react';

    // Pass the initialized instance
    export function VideoFeed({ shortKit }) {
      return (
        <div style={{ width: '100%', height: '100vh' }}>
          <ShortKitFeed shortKit={shortKit} />
        </div>
      );
    }
    ```

    Partial-height feed:

    ```tsx PartialFeed.tsx
    import React from 'react';
    import { ShortKitFeed } from '@shortkit/web/react';

    export function PartialFeed({ shortKit }) {
      return (
        <div className="feed-container">
          <ShortKitFeed
            shortKit={shortKit}
            feedHeight={{ percentage: 0.7 }}
          />
        </div>
      );
    }
    ```

    ```css
    .feed-container {
      width: 400px;
      height: 80vh;
      margin: 0 auto;
    }
    ```
  </Tab>
  <Tab title="Vanilla JS">
    Mount the feed to any DOM element:

    ```typescript
    import { ShortKit } from '@shortkit/web';

    const shortKit = ShortKit.init({
      apiKey: 'pk_live_your_publishable_key',
      config: { feedHeight: 'fullscreen' },
    });

    const container = document.getElementById('video-feed');
    const feed = shortKit.mount(container);

    // Later, to unmount:
    feed.unmount();
    ```

    ```html
    <div id="video-feed" style="width: 100%; height: 100vh;"></div>
    ```
  </Tab>
</Tabs>

### Responsive layout

The feed automatically adapts to its container size. For a mobile-like experience on desktop, constrain the container:

```css
.mobile-container {
  max-width: 430px;
  height: 100vh;
  margin: 0 auto;
}

@media (max-width: 430px) {
  .mobile-container {
    max-width: 100%;
  }
}
```

## Configuration

### FeedConfig

```typescript
interface FeedConfig {
  feedHeight?: 'fullscreen' | { percentage: number };
  controls?: ControlsConfig;
  theme?: ThemeConfig;
}
```

### ControlsConfig

```typescript
interface ControlsConfig {
  playPause?: boolean;        // default: true
  scrubber?: boolean;         // default: true
  captions?: CaptionConfig;   // default: { enabled: true, defaultOn: false }
  volume?: boolean;           // default: true
  muteOnStart?: boolean;      // default: true
  share?: ShareConfig;        // default: { enabled: true }
  autoHideControls?: AutoHideConfig; // default: { enabled: true, delayMs: 3000 }
}

interface CaptionConfig {
  enabled?: boolean;          // default: true
  defaultOn?: boolean;        // default: false
  style?: CaptionStyle;
}

interface ShareConfig {
  enabled?: boolean;          // default: true
}

interface AutoHideConfig {
  enabled?: boolean;          // default: true
  delayMs?: number;           // default: 3000
}
```

### ThemeConfig

```typescript
interface ThemeConfig {
  primaryColor?: string;              // default: '#FFFFFF'
  backgroundColor?: string;           // default: '#000000'
  controlsTint?: string;              // default: '#FFFFFF'
  controlsBackgroundOpacity?: number;  // default: 0.4
  fontFamily?: string;                 // default: 'system-ui'
  scrubberStyle?: ScrubberStyle;
  captionStyle?: CaptionStyle;
}

interface ScrubberStyle {
  height?: number;             // default: 3
  knobSize?: number;           // default: 12
  playedColor?: string;        // default: '#FFFFFF'
  bufferedColor?: string;      // default: 'rgba(255,255,255,0.5)'
  unplayedColor?: string;      // default: 'rgba(255,255,255,0.3)'
}

interface CaptionStyle {
  fontSize?: number;           // default: 16
  textColor?: string;          // default: '#FFFFFF'
  backgroundColor?: string;    // default: '#000000'
  backgroundOpacity?: number;  // default: 0.7
  position?: 'top' | 'bottom'; // default: 'bottom'
}
```

<Tip>
  All config fields are optional. `ShortKit.init({ apiKey })` with just an API key and empty config produces a working feed with sensible defaults.
</Tip>

### Remote configuration

The SDK fetches a remote configuration overlay on initialization. Server-side overrides merge on top of your code-level defaults, enabling A/B testing and remote control without redeployment.

Configuration resolves in this order:

1. Your code-level config renders the feed immediately.
2. A remote overlay loads in the background (2-second timeout, non-blocking).
3. Server-provided fields override your defaults. Fields not included in the overlay keep your values.
4. The overlay is cached in localStorage with a configurable TTL (default 5 minutes).

## User identity

ShortKit generates an anonymous ID on first initialization (stored in localStorage). All engagement tracking and feed personalization work immediately without requiring a user ID.

When your user logs in, associate their identity for cross-device personalization:

<Tabs>
  <Tab title="React">
    ```tsx
    import { ShortKit } from '@shortkit/web';

    // After your authentication completes
    shortKit.setUserId('user_12345');

    // On logout -- generates a new anonymous ID
    shortKit.clearUserId();
    ```
  </Tab>
  <Tab title="Vanilla JS">
    ```typescript
    // After your authentication completes
    shortKit.setUserId('user_12345');

    // On logout -- generates a new anonymous ID
    shortKit.clearUserId();
    ```
  </Tab>
</Tabs>

<Note>
  Calling `setUserId` triggers backend identity resolution, merging the anonymous session's engagement data with the identified user profile. Call `clearUserId` when the user logs out.
</Note>

## Keyboard shortcuts

The feed supports keyboard navigation when focused:

| Key | Action |
|-----|--------|
| `Space` | Play / Pause |
| `Up` / `Down` | Previous / Next video |
| `Left` / `Right` | Seek backward / forward 5 seconds |
| `M` | Mute / Unmute |
| `F` | Toggle fullscreen |
| `C` | Toggle captions |

Disable keyboard handling if it conflicts with your app:

<Tabs>
  <Tab title="React">
    ```tsx
    <ShortKitFeed shortKit={shortKit} keyboardEnabled={false} />
    ```
  </Tab>
  <Tab title="Vanilla JS">
    ```typescript
    shortKit.mount(container, { keyboardEnabled: false });
    ```
  </Tab>
</Tabs>

## Server-side rendering

The SDK is client-side only. For SSR frameworks, use dynamic imports to prevent the SDK from loading on the server.

### Next.js

```tsx VideoFeed.tsx
'use client';

import dynamic from 'next/dynamic';

const ShortKitFeed = dynamic(
  () => import('@shortkit/web/react').then(mod => mod.ShortKitFeed),
  { ssr: false }
);

export function VideoFeed({ shortKit }) {
  return <ShortKitFeed shortKit={shortKit} />;
}
```

Initialize in a client component:

```tsx providers.tsx
'use client';

import { useEffect, useRef } from 'react';
import { ShortKit } from '@shortkit/web';

export function ShortKitClientProvider({ children }: { children: React.ReactNode }) {
  const shortKitRef = useRef<ReturnType<typeof ShortKit.init> | null>(null);

  useEffect(() => {
    if (!shortKitRef.current) {
      shortKitRef.current = ShortKit.init({
        apiKey: 'pk_live_your_publishable_key',
        config: {
          feedHeight: 'fullscreen',
          theme: {
            primaryColor: '#FFFFFF',
            backgroundColor: '#000000',
          },
        },
      });
    }
  }, []);

  return children;
}
```

### Remix / Other SSR frameworks

Use a similar pattern -- guard initialization behind a `useEffect` or `typeof window !== 'undefined'` check:

```typescript
if (typeof window !== 'undefined') {
  const shortKit = ShortKit.init({ apiKey: '...', config: { /* ... */ } });
}
```

## Environment

Control which ShortKit environment the SDK connects to:

```typescript
const shortKit = ShortKit.init({
  apiKey: 'pk_live_your_publishable_key',
  config: { /* ... */ },
  environment: 'production', // default
});
```

| Environment | Description |
|---|---|
| `'production'` | Live environment. Use in production builds. |
| `'staging'` | Pre-release testing environment. |
| `'development'` | Local development. Enables debug logging in the browser console. |

## Content Security Policy

If your site uses CSP headers, allow these sources:

```
script-src 'self' https://cdn.shortkit.dev;
connect-src 'self' https://api.shortkit.dev https://*.shortkit.dev;
media-src 'self' blob: https://*.shortkit.dev;
img-src 'self' data: https://*.shortkit.dev;
```
